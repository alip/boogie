<%doc>
    Template for play subcommand.
    Called before mpd.play() function is called with function arguments.
    pos is playlist position.
</%doc>\
<%
    from boogie.colour import bold, green, yellow
%>\
\
%if pos is None:
${yellow(_("Playing..."))}
%else:
<% from boogie.colour import colourify %>\
${colourify(
    _("Playing song at position %(pos)s..."), yellow, pos=bold(green(str(pos+1)))
    )
}
%endif

<%doc>
    Template for songs.
    Used by subcommands: currentsong, status
    Example output:
    Syd Barrett - Boogie
    Album: Have You Got It Yet? Genre: Psychedelic Rock
</%doc>\
<%
    from boogie.colour import bold, cyan, magenta, yellow

%>\
\
%if "title" in song:
${cyan(song.get("artist", "???"))} - ${cyan(song["title"])}
%elif "file" in song:
${cyan(song["file"])}
%endif
\
<% nl = False %>\
%if "album" in song:
${yellow(_("Album"))}: ${bold(magenta(song["album"]))}\
<% nl = True %>\
%endif
\
%if "genre" in song:
 ${yellow(_("Genre"))}: ${bold(magenta(song["genre"]))}
%elif nl:

%endif

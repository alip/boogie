<%doc>
    Template for stats subcommand.
    Example output:
    Artists: 284
    Albums:  410
    Songs:   2033

    Play Time:  5 days, 15:19:29
    Uptime:     0 days, 07:41:59
    DB Updated: Sat Jul 19 14:31:37 2008
</%doc>\
<%
    from time import ctime
    from boogie.colour import cyan, green, magenta, yellow
    from boogie.util import asHuman, fillSpace

    artists, albums, songs =\
        fillSpace(_("Artists") + ":", _("Albums") + ":", _("Songs") + ":")
    playtime, uptime, db_updated =\
        fillSpace(_("Play Time") + ":", _("Uptime") + ":", _("DB Updated") + ":")
%>\
\
${yellow(artists)} ${green(stats["artists"])}
${yellow(albums)} ${green(stats["albums"])}
${yellow(songs)} ${green(stats["songs"])}

${yellow(playtime)} ${green(asHuman(stats["db_playtime"]))}
${yellow(uptime)} ${green(asHuman(stats["uptime"]))}
${yellow(db_updated)} ${green(ctime(int(stats["db_update"])))}
